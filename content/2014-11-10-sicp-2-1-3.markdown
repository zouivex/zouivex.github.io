```
	(* (exp 2 a)
	   (exp 3 b)))

	(log2 (mod z 3))
```

## 练习2.6

``` Scheme
(define zero (lambda(f) (lambda(x) x)))

(define (add-1 n)
  (lambda(f) (lambda(x) (f (n f) x))))

(define one
  (lambda (f)
    (lambda (x) x)))

(define two
  (lambda (f)
    (lambda (x) (f (f x)))))

(define (add a b)
  (lambda (f)
    (lambda (x)
      ((a f) ((b f) x)))))
```
