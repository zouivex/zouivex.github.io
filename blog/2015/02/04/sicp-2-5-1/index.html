练习2.77

开始时z的值为(complex (rectangular (3 4)))。第一次调用时，apply-generic会根据标签complex将其分派为过程调用(magnitude '(rectangular (3 4))。第二次调用时，apply-generic会根据标签rectangular将其分派为rectangular包中所定义的magnitude过程。

练习2.78

(define (attach-tag type-tag contents)
  (if (number? contents)
      contents
      (cons type-tag contents)))

(define (type-tag data)
  (cond ((number? data) 'scheme-number)
        ((pair? data) (car data))
        (else (error "Bad tagged datum -- TYPE-TAG" data))))

(define (contents data)
  (cond ((number? data) data)
        ((pair? data) (cdr data))
        (else (error "Bad tagged dataum -- TYPE-TAG" data))))

练习2.79

(put 'equ? '(scheme-number scheme-number)
       (lambda (x y)
         (= x y)))

(put 'equ? '(rational rational)
       (lambda (x y)
         (and (= (numer x) (numer y))
              (= (denom x) (denom y)))))

(put 'equ? '(complex complex)
       (lambda (z1 z2)
         (and (= (real-part z1) (real-part z2))
              (= (imag-part z1) (imag-part z2)))))

(define (equ? x y)
  (apply-generic 'equ? x y))

练习2.80

(put '=zero? '(scheme-number)
       (lambda (x)
         (= x 0)))

(put '=zero? '(rational)
       (lambda (x)
         (= (numer x) 0)))

(put '=zero? '(complex)
       (lambda (x)
         (and (= (real-part x) 0)
              (= (imag-part x) 0))))

(define (=zero? x)
  (apply-generic '=zero? x))
