采用替换模型进行分析:

(define (f g) (g 2))

如果需要求值:

(f f)

采用替换模型替换为:

(f 2)

进一步替换为:

(2 2)

由于2不是一个可以调用的过程，最终求值(f f)会导致运行时错误。

(define (fixed-point f first-guest)
  (define (close-enough? v1 v2)
    (< (abs (- v1 v2)) tolerance))
  (define (try guess)
    (display guess)
    (newline)
    (let ((next (f guess))
          )
      (if (close-enough? guess next)
          next
          (try next))))
  (try first-guest))
  • 练习1.37
(define (cont-frac n d k)
  (if (= k 1)
      (/ (n k) (d k))
      (/ (n k) (+ (d k) (cont-frac n d (- k 1))))))

(define (cont-frac-iter n d k)
  (define (iter k result)
    (if (= k 1)
        result
        (iter (- k 1)
              (/ (n k) (+ (d k) result)))))
  (iter k 0))
